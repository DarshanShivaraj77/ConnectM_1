🗄️ Complete Database Schema

Core Tables + RLS + Materialized Views

-- DUAL GROUP SYSTEM
CREATE TYPE group_type AS ENUM ('hierarchy', 'flat');
CREATE TYPE group_domain AS ENUM ('assets', 'routes', 'hubs', 'geofence', 'service_centers');

CREATE TABLE groups (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255),
    org_id INTEGER REFERENCES organizations(id),
    type group_type,
    domain group_domain,
    is_physical BOOLEAN DEFAULT FALSE, -- Hub = physical
    parent_id INTEGER REFERENCES groups(id),
    level INTEGER CHECK (level >= 0 AND level <= 6),
    path LTREE, -- ltree hierarchy
    template_id INTEGER REFERENCES group_templates(id),
    created_by INTEGER REFERENCES users(id)
);

-- PHYSICAL OWNERSHIP (1:1)
CREATE TABLE physical_asset_mapping (
    asset_id INTEGER PRIMARY KEY REFERENCES assets(id),
    hub_group_id INTEGER NOT NULL REFERENCES groups(id),
    mapped_at TIMESTAMP DEFAULT NOW()
);

-- LOGICAL STAKEHOLDER INTERESTS (N:N)
CREATE TABLE stakeholder_asset_groups (
    asset_id INTEGER REFERENCES assets(id),
    group_id INTEGER REFERENCES groups(id),
    stakeholder_role VARCHAR(50), -- OEM, Financier, Fleet, 3PL, Driver
    interest_level VARCHAR(20) DEFAULT 'read',
    PRIMARY KEY (asset_id, group_id)
);
